int solution(vector<int> &A) {
    // write your code in C++14 (g++ 6.2.0)
    int A_size = A.size();
    long count = 0;
    int add_base = 0;
    for(int i = 0; i<A_size; i++)
    {
        if(A[i] == 0)
        {
            add_base++;
        }
        else 
        {
            count += add_base;
        }
    }
    if(count > 1000000000)
        return -1;
    return count;
}
